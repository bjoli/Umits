kilo[x] = {x} * 1000
deci[x] = {x} / 10
centi[x] = {x} / 100
milli[x] = {x} / 1000
micro[x] = {x} / 1000000
nano[x] = {x} / 1e9

// Power (Reference: 1 Watt and 1 milliWatt)
dBW[power] = 10 * log10({power} / 1W)
dBm[power] = 10 * log10({power} / 1mW)

// Voltage (Reference: 1 Volt)
dBV[voltage] = 20 * log10({voltage} / 1V)

// Acoustics (Reference: 20 micropascals)
dBSPL[pressure] = 20 * log10({pressure} / 20uPa)
